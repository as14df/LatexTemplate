\newcommand{\initializeTitle}{
	\clearpage
	\setcounter{page}{0}
}

\newcommand{\initializePreambel}{
	\clearpage
	% Set footer
	\cfoot[]{}
	\ofoot[\pagemark]{\pagemark}
	% Set roman page numbering
	\pagenumbering{roman} 
	\normalfont	
}

\newcommand{\initializeMain}{
	\clearpage
	% Set header
	\renewcommand*{\chaptermarkformat}{}
	\automark{chapter}
	\ohead{\thechapter~\headmark}
	\chead[]{}
	% Set arabic page numbering
	\newcounter{savepage}
	\setcounter{savepage}{\arabic{page}}
	\pagenumbering{arabic}
}

\newcommand{\initializeAppendix}{
	\clearpage
	% Set roman page numbering
	\pagenumbering{roman}
	\setcounter{page}{\thesavepage}
	\appendix
    \addtocontents{toc}{\protect\contentsline{chapter}{Anhang \protect\numberline{}}{}{}}
}

\newcommand{\initializeBibliography}{
	\clearpage

	\cleardoublepage
}

\newcommand{\initializeIndex}{
	\clearpage
	\addcontentsline{toc}{chapter}{Index}
